import cv2 
import numpy as np
from matplotlib import pyplot as plt

img = cv2.imread("figura_1_BW.png", cv2.IMREAD_COLOR)
img_hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV) # Seems to work better in HSV space 

vectorized = img_hsv.reshape((-1,3)) # num_pixels x channel array for kmeans()
vectorized = vectorized.astype(np.float32) # kmeans needs float32 values

print(vectorized.shape)

# K-means
K = 2
criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 10, 1.0)

# Reconstruct original image
center = np.uint8(center)
res = center[label.flatten()]

import cv2
import numpy as np
from sklearn.cluster import MeanShift
from matplotlib import pyplot as plt

image = cv2.imread("figura_1_BW.png", 0)
image = cv2.medianBlur(image, 3)

X = np.float32(image.reshape((-1,1)))

ms = MeanShift()
ms.fit(X)

labels = ms.predict(X)
segmented_img = labels.reshape(image.shape)

plt.subplot(1, 2, 1)
plt.imshow(image, cmap='gray')
plt.title('Original Image')

plt.title('Segmented Image')

import cv2
import numpy as np
from sklearn.cluster import MeanShift, estimate_bandwidth


image = cv2.imread("fieldWheat.png", 0)
image = cv2.medianBlur(image, 3)

X = np.float32(image.reshape((-1,1)))

bandwidth = estimate_bandwidth(X, quantile=0.25, n_samples=50)
ms = MeanShift(bandwidth=bandwidth, bin_seeding=True)
ms.fit(X)

labels = ms.predict(X)

segmented_img = labels.reshape(image.shape)
plt.subplot(1, 2, 1)
plt.imshow(image, cmap='gray')
plt.title('Original Image')
plt.subplot(1, 2, 2)
plt.imshow(segmented_img, cmap='gray')
plt.title('Segmented Image')
plt.show()
